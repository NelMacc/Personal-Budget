app.use(express.json());

let budget = 0;
let expenses = [];

// Route to set budget
app.put('/budget', (req, res) => {
    const { amount } = req.body;
    budget = amount;
    res.json({ success: true, budget });
});

// Route to add an expense
app.post('/expenses', (req, res) => {
    const { amount, description } = req.body;
    if (amount > budget) {
        return res.status(400).json({ error: "Expense exceeds budget" });
    }
    expenses.push({ amount, description });
    budget -= amount;
    res.json({ success: true, budget, expenses });
});

// Route to get current budget and expenses
app.get('/status', (req, res) => {
    res.json({ budget, expenses });
});

envelopesRouter.put('/envelopes/:id', (req, res) => {
  const envelope = envelopes.find(e => e.id === parseInt(req.params.id));
  if (!envelope) res.status(404).send('The envelope with that ID was not found.');

  const { category, budget } = req.body;

  if (!category && !budget) {
    res.status(404).send('Updated category and budget were not found');
    return;
  } 

  if (category) {
    envelope.category = category;
  }
  if (budget) {
    envelope.budget = budget;
  }
  
  res.send(envelope);
});
